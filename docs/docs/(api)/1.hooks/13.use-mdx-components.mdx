# useMdxComponents()

Returns the component map registered via `docs/mdx-components.tsx`. Use this in custom layouts to access user-defined component overrides.

```tsx
import { useMdxComponents } from 'boltdocs/client'
```

---

## Return Value

```tsx
const components: Record<string, React.ComponentType> = useMdxComponents()
```

---

## Usage Example

```tsx
import { useMdxComponents, CopyMarkdown } from 'boltdocs/client'

function ContentHeader({ mdxRaw, route, config }) {
  const components = useMdxComponents()

  // Use the user's CopyMarkdown override, or fall back to the default
  const CopyBtn = (components.CopyMarkdown as any) || CopyMarkdown

  return (
    <div className="flex items-center justify-between">
      <Breadcrumbs />
      <CopyBtn mdxRaw={mdxRaw} route={route} config={config} />
    </div>
  )
}
```
