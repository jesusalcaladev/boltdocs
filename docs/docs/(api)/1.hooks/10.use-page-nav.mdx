# usePageNav()

Returns the previous and next pages relative to the current route. Respects tab boundaries — if the user is in the "Guides" tab, navigation stays within that tab.

```tsx
import { usePageNav } from 'boltdocs/hooks'
```

---

## Return Value

```tsx
const {
  prevPage,       // ComponentRoute | null
  nextPage,       // ComponentRoute | null
  currentRoute,   // ComponentRoute | null
} = usePageNav()
```

---

## Usage Example

```tsx
import { usePageNav } from 'boltdocs/hooks'

function PageNavigation() {
  const { prevPage, nextPage } = usePageNav()

  return (
    <div className="flex justify-between mt-12 pt-6 border-t">
      {prevPage ? (
        <a href={prevPage.path}>← {prevPage.title}</a>
      ) : <span />}
      {nextPage ? (
        <a href={nextPage.path}>{nextPage.title} →</a>
      ) : <span />}
    </div>
  )
}
```
