# useTabs(tabs, routes)

Manages the navigation tab state, including the active index and an animated indicator for the selected tab.

```tsx
import { useTabs } from 'boltdocs/hooks'
```

---

## Parameters

| Parameter | Type | Description |
| :--- | :--- | :--- |
| `tabs` | `BoltdocsTab[]` | Tab definitions from config |
| `routes` | `ComponentRoute[]` | Routes to determine which tab is active |

---

## Return Value

```tsx
const {
  tabs,            // BoltdocsTab[] — the input tabs
  activeIndex,     // number — index of the currently active tab
  activeTabId,     // string | undefined — ID of the active tab (e.g., 'guides')
  indicatorStyle,  // CSSProperties — position/size for the animated underline
  tabRefs,         // MutableRefObject<(HTMLAnchorElement | null)[]> — refs for tab elements
} = useTabs(tabs, routes)
```

---

## Usage Example

```tsx
import { useTabs, useRoutes, useConfig } from 'boltdocs/hooks'

function TabBar() {
  const config = useConfig()
  const { routes } = useRoutes()
  const tabs = config.theme?.tabs || []
  const { activeIndex, indicatorStyle, tabRefs } = useTabs(tabs, routes)

  return (
    <nav className="relative flex border-b">
      {tabs.map((tab, i) => (
        <a
          key={tab.id}
          ref={(el) => { tabRefs.current[i] = el }}
          className={`px-4 py-2 ${i === activeIndex ? 'text-primary-500' : 'text-zinc-400'}`}
        >
          {typeof tab.text === 'string' ? tab.text : tab.text.en}
        </a>
      ))}
      <div
        className="absolute bottom-0 h-0.5 bg-primary-500 transition-all duration-300"
        style={indicatorStyle}
      />
    </nav>
  )
}
```
