# useOnThisPage(headings)

Provides state for the table of contents component, including the list of headings and the currently-active heading ID.

```tsx
import { useOnThisPage } from 'boltdocs/hooks'
```

---

## Parameters

| Parameter | Type | Description |
| :--- | :--- | :--- |
| `headings` | `Array<{ id: string; text: string; level: number }>` | Headings extracted from the current page |

---

## Return Value

```tsx
const {
  headings,     // The input headings array
  activeId,     // string | null — ID of the heading currently in view
  setActiveId,  // (id: string | null) => void
} = useOnThisPage(headings)
```

---

## Usage Example

```tsx
import { useOnThisPage, useRoutes } from 'boltdocs/hooks'

function TableOfContents() {
  const { currentRoute } = useRoutes()
  const { headings, activeId } = useOnThisPage(currentRoute?.headings)

  return (
    <ul>
      {headings.map((h) => (
        <li
          key={h.id}
          className={h.id === activeId ? 'text-primary-500' : 'text-zinc-400'}
          style={{ paddingLeft: `${(h.level - 2) * 12}px` }}
        >
          <a href={`#${h.id}`}>{h.text}</a>
        </li>
      ))}
    </ul>
  )
}
```
